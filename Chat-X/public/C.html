Module 2: Data Types, Variables, and Operators

2.1 Data Types

Basic data types in C include int, float, double, and char...

#include <stdio.h>

int main() {
    int age = 30;
    float height = 5.9;
    double pi = 3.141592;
    char initial = 'J';

    printf("Age: %d\\n", age);
    printf("Height: %.1f\\n", height);
    printf("Pi: %.5f\\n", pi);
    printf("Initial: %c\\n", initial);

    return 0;
}

2.2 Operators

Operators in C include arithmetic, relational, logical, and bitwise operators...

#include <stdio.h>

int main() {
    int a = 10, b = 20;
    printf("a + b = %d\\n", a + b);
    printf("a - b = %d\\n", a - b);
    printf("a * b = %d\\n", a * b);
    printf("a / b = %d\\n", a / b);
    printf("a %% b = %d\\n", a % b);

    return 0;
}

Module 3: Control Structures

3.1 Conditional Statements

If-Else statements help in making decisions in C programming...

#include <stdio.h>

int main() {
    int number = 15;

    if (number % 2 == 0) {
        printf("The number is even.\\n");
    } else {
        printf("The number is odd.\\n");
    }

    return 0;
}

3.2 Loops

Loops allow you to execute a block of code multiple times...

#include <stdio.h>

int main() {
    // For loop
    for (int i = 0; i < 5; i++) {
        printf("For loop iteration %d\\n", i);
    }

    // While loop
    int j = 0;
    while (j < 5) {
        printf("While loop iteration %d\\n", j);
        j++;
    }

    // Do-While loop
    int k = 0;
    do {
        printf("Do-While loop iteration %d\\n", k);
        k++;
    } while (k < 5);

    return 0;
}

Module 4: Functions

4.1 Function Basics

Functions in C help in organizing code into reusable blocks...

#include <stdio.h>

void greet() {

4.2 Function Parameters and Return Values

Functions can accept parameters and return values...

#include <stdio.h>

int add(int a, int b) {
    return a + b;
}

int main() {
    int result = add(5, 3);

Module 5: Arrays & Strings

5.1 Arrays

Arrays store multiple values of the same type in a single variable...

#include <stdio.h>

int main() {
    int numbers[5] = {1, 2, 3, 4, 5}; // Array of integers

    for (int i = 0; i < 5; i++) {
        printf("Number %d: %d\\n", i, numbers[i]); // Accessing each element
    }

    return 0;
}

5.2 Strings

Strings in C are arrays of characters terminated by a null character (`'\0'`)...

#include <stdio.h>
#include <string.h>

int main() {
    char name[] = "John Doe"; // String initialization

    printf("Name: %s\\n", name); // Printing string

    // String length
    printf("Length of the name: %zu\\n", strlen(name));

    return 0;
}

Module 6: Pointers & Memory

6.1 Pointers

Pointers are variables that store the address of another variable...

#include <stdio.h>

int main() {
    int value = 10;      // Integer variable
    int *ptr = &value;   // Pointer to integer

    printf("Value: %d\\n", value);
    printf("Pointer: %p\\n", ptr);           // Address of the variable
    printf("Value through pointer: %d\\n", *ptr); // Value at the address

    return 0;
}

6.2 Dynamic Memory Allocation

Dynamic memory allocation allows you to allocate memory at runtime...

#include <stdio.h>
#include <stdlib.h>

int main() {
    int *arr = (int *)malloc(5 * sizeof(int)); // Allocating memory for an array of 5 integers

    for (int i = 0; i < 5; i++) {
        arr[i] = i * 2; // Initializing array
        printf("Element %d: %d\\n", i, arr[i]);
    }

    free(arr); // Deallocating memory

    return 0;
}

Module 7: File I/O

7.1 Reading from Files

Files can be read using functions like fopen, fgetc, and fgets...

#include <stdio.h>

int main() {
    FILE *file = fopen("example.txt", "r");
    if (file == NULL) {
        printf("Failed to open file.\\n");
        return 1;
    }

    char ch;
    while ((ch = fgetc(file)) != EOF) {
        printf("%c", ch);
    }

    fclose(file);
    return 0;
}

7.2 Writing to Files

Files can be written using functions like fopen, fprintf, and fputs...

#include <stdio.h>

int main() {
    FILE *file = fopen("example.txt", "w");
    if (file == NULL) {
        printf("Failed to open file.\\n");
        return 1;
    }

    fprintf(file, "Hello, World!\\n");
    fclose(file);

    return 0;
}

Module 8: Structures

8.1 Defining and Using Structures

Structures are used to group variables of different types under a single name...

#include <stdio.h>

struct Person {
    char name[50];
    int age;
};

int main() {
    struct Person person;
    strcpy(person.name, "Alice");
    person.age = 30;

    printf("Name: %s\\n", person.name);
    printf("Age: %d\\n", person.age);

    return 0;
}

Module 9: Dynamic Memory Allocation

9.1 Allocating and Deallocating Memory

Dynamic memory allocation allows you to allocate and deallocate memory at runtime...

#include <stdio.h>
#include <stdlib.h>

int main() {
    int *arr = (int *)malloc(10 * sizeof(int));
    if (arr == NULL) {
        printf("Memory allocation failed.\\n");
        return 1;
    }

    for (int i = 0; i < 10; i++) {
        arr[i] = i;
        printf("%d\\n", arr[i]);
    }

    free(arr);
    return 0;
}
