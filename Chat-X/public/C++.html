9.2 Virtual Functions and Polymorphism

Implementing polymorphism using virtual functions...

#include 
using namespace std;

class Base {
public:
    virtual void show() {
        cout << "Base class" << endl;
    }
};

class Derived : public Base {
public:
    void show() override {
        cout << "Derived class" << endl;
    }
};

int main() {
    Base *ptr;
    Derived obj;
    ptr = &obj;

    ptr->show();

    return 0;
}

Module 10: Templates and Standard Template Library (STL)

10.1 Function Templates

Creating and using function templates...

using namespace std;

template 
T max(T a, T b) {
    return (a > b) ? a : b;
}

int main() {
    cout << "Max of 10 and 20: " << max(10, 20) << endl;
    cout << "Max of 3.5 and 2.5: " << max(3.5, 2.5) << endl;

    return 0;
}

10.2 Class Templates

Creating and using class templates...

#include 
using namespace std;

template 
class Stack {
private:
    T arr[100];
    int top;

public:
    Stack() : top(-1) {}

    void push(T value) {
        arr[++top] = value;
    }

    T pop() {
        return arr[top--];
    }
};

int main() {
    Stack intStack;
    intStack.push(1);
    cout << "Popped: " << intStack.pop() << endl;

    return 0;
}
