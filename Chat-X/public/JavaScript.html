Module 1: Introduction to JavaScript

1.1 Overview of JavaScript

JavaScript is a versatile programming language used primarily for creating dynamic and interactive web pages...

// Displaying a message in the console
console.log('Hello, World!');

1.2 Setting up the Environment

JavaScript can be written directly in HTML files using the <script> tag or in separate .js files...

Module 2: Variables and Data Types

2.1 Understanding Variables

Variables are used to store data values...

// Declaring variables
let name = 'John';
const age = 30;
var isStudent = true;

console.log(name);
console.log(age);
console.log(isStudent);

2.2 Primitive Data Types

JavaScript has several primitive data types: string, number, boolean, undefined, and null...

// Type conversion
let num = '25';
num = Number(num); // Convert string to number
console.log(num + 5); // Output: 30

Module 3: Operators

3.1 Arithmetic Operators

Arithmetic operators are used to perform mathematical operations...

// Arithmetic operators
let x = 10;
let y = 5;

console.log(x + y); // Addition
console.log(x - y); // Subtraction
console.log(x * y); // Multiplication
console.log(x / y); // Division
console.log(x % y); // Modulus

3.2 Comparison Operators

Comparison operators are used to compare values...

// Comparison operators
let a = 10;
let b = 20;

console.log(a == b); // Equal to
console.log(a != b); // Not equal to
console.log(a > b);  // Greater than
console.log(a < b);  // Less than

3.3 Logical Operators

Logical operators are used to combine boolean values...

// Logical operators
let p = true;
let q = false;

console.log(p && q); // Logical AND
console.log(p || q); // Logical OR
console.log(!p);     // Logical NOT

Module 4: Control Structures

4.1 Conditional Statements

Conditional statements are used to perform different actions based on different conditions...

// if...else statement
let score = 85;

if (score >= 90) {
    console.log('A');
} else if (score >= 80) {
    console.log('B');
} else {
    console.log('C');
}

4.2 Loops

Loops are used to execute a block of code repeatedly...

// for loop
for (let i = 0; i < 5; i++) {
    console.log(i);
}

// while loop
let j = 0;
while (j < 5) {
    console.log(j);
    j++;
}

// do...while loop
let k = 0;
do {
    console.log(k);
    k++;
} while (k < 5);

Module 5: Functions

5.1 Function Declaration and Expression

Functions are blocks of code designed to perform a particular task...

// Function declaration
function greet(name) {
    return 'Hello, ' + name;
}

console.log(greet('Alice'));

// Function expression
const add = function(a, b) {
    return a + b;
};

console.log(add(5, 10));

5.2 Arrow Functions

Arrow functions provide a more concise syntax for writing function expressions...

// Arrow function
const multiply = (a, b) => a * b;

console.log(multiply(5, 10));

5.3 Callback Functions

Callback functions are functions passed as arguments to other functions...

// Callback function
function processData(data, callback) {
    console.log('Processing data...');
    callback(data);
}

processData('Sample data', function(data) {
    console.log('Callback executed with data:', data);
});

Module 6: Objects and Arrays

6.1 Creating and Using Objects

Objects are collections of key-value pairs...

// Creating an object
const person = {
    name: 'John',
    age: 30,
    greet: function() {
        console.log('Hello, ' + this.name);
    }
};

console.log(person.name);
person.greet();

6.2 Working with Arrays

Arrays are ordered collections of values...

// Array methods
const numbers = [1, 2, 3, 4, 5];

console.log(numbers.length); // Length of the array
numbers.push(6); // Add an element
console.log(numbers);

numbers.pop(); // Remove the last element
console.log(numbers);

numbers.forEach(number => console.log(number)); // Iterate through elements

Module 7: DOM Manipulation

7.1 Selecting Elements

The Document Object Model (DOM) represents the structure of a web page...

// Selecting elements
const title = document.getElementById('title');
const paragraphs = document.getElementsByTagName('p');
const items = document.querySelectorAll('.item');

console.log(title);
console.log(paragraphs);
console.log(items);

7.2 Modifying Elements

Manipulating DOM elements allows you to change the content and style of a page...

// Modifying elements
const header = document.querySelector('h1');
header.textContent = 'New Title';
header.style.color = 'blue';

7.3 Event Handling

Event handling allows you to execute code in response to user interactions...

// Event handling
const button = document.querySelector('button');

button.addEventListener('click', function() {
    alert('Button clicked!');
});

Module 8: Asynchronous JavaScript

8.1 Callbacks

Callbacks are functions passed as arguments to other functions...

// Callback function
function fetchData(callback) {
    setTimeout(() => {
        callback('Data received');
    }, 1000);
}

fetchData(data => console.log(data));

8.2 Promises

Promises represent the eventual completion (or failure) of an asynchronous operation...

// Promise
const promise = new Promise((resolve, reject) => {
    setTimeout(() => {
        resolve('Data received');
    }, 1000);
});

promise.then(data => console.log(data));

8.3 Async/Await

Async/await provides a more readable way to handle asynchronous code...

// Async/await
async function fetchData() {
    const data = await new Promise((resolve, reject) => {
        setTimeout(() => {
            resolve('Data received');
        }, 1000);
    });
    console.log(data);
}

fetchData();

Module 9: Error Handling

9.1 Try...Catch

Error handling is crucial for robust code...

// Try...catch
try {
    const result = riskyFunction();
    console.log(result);
} catch (error) {
    console.error('Error:', error.message);
}

function riskyFunction() {
    throw new Error('Something went wrong');
}

9.2 Throwing Errors

You can throw your own errors using the throw statement...

// Throwing errors
function validateAge(age) {
    if (age < 0) {
        throw new Error('Age cannot be negative');
    }
    return age;
}

console.log(validateAge(25));
console.log(validateAge(-5)); // This will throw an error

Module 10: Modules

10.1 Importing and Exporting Modules

Modules allow you to split your code into separate files...

// module.js
export const greeting = 'Hello';
export function greet(name) {
    return `${greeting}, ${name}!`;
}

// main.js
import { greeting, greet } from './module.js';

console.log(greeting);
console.log(greet('Alice'));

10.2 Using ES6 Modules

ES6 modules help in organizing your code...

// Using ES6 modules
import * as math from './math.js';

console.log(math.add(2, 3));
console.log(math.subtract(5, 2));
