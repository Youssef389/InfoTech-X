Module 2: Variables & Data Types

2.1 Variables

Variables in Go are declared using the var keyword...

package main

import "fmt"

func main() {
    var age int = 30
    var height float64 = 5.9
    var name string = "John"

    fmt.Println("Age:", age)
    fmt.Println("Height:", height)
    fmt.Println("Name:", name)
}

2.2 Data Types

Go supports several basic data types, including int, float64, bool, and string...

package main

import "fmt"

func main() {
    var isTrue bool = true
    var pi float64 = 3.14159

    fmt.Println("Boolean:", isTrue)
    fmt.Println("Pi:", pi)
}

Module 5: Arrays & Slices

5.1 Arrays

Arrays in Go are fixed-size collections of elements...

package main

import "fmt"

func main() {
    var numbers [5]int = [5]int{1, 2, 3, 4, 5}

    for i, number := range numbers {
        fmt.Println("Number", i, ":", number)
    }
}

5.2 Slices

Slices are more flexible than arrays and are used for dynamic-sized collections...

package main

import "fmt"

func main() {
    var numbers []int = []int{1, 2, 3, 4, 5}

    for i, number := range numbers {
        fmt.Println("Number", i, ":", number)
    }
}

Module 6: Structs & Interfaces

6.1 Structs

Structs are used to define custom types with multiple fields...

package main

import "fmt"

type Person struct {
    Name string
    Age  int
}

func main() {
    person := Person{Name: "John", Age: 30}
    fmt.Println("Name:", person.Name)
    fmt.Println("Age:", person.Age)
}

6.2 Interfaces

Interfaces in Go are used to specify methods that a type must implement...

package main

import "fmt"

type Greeter interface {
    Greet() string
}

type Person struct {
    Name string
}

func (p Person) Greet() string {
    return "Hello, " + p.Name
}

func main() {
    var greeter Greeter = Person{Name: "John"}
    fmt.Println(greeter.Greet())
}

Module 7: Concurrency

7.1 Goroutines

Goroutines are lightweight threads managed by Go runtime...

package main

import (
    "fmt"
    "time"
)

func printNumbers() {
    for i := 1; i <= 5; i++ {
        fmt.Println(i)
        time.Sleep(time.Second)
    }
}

func main() {
    go printNumbers()
    time.Sleep(6 * time.Second)
}

7.2 Channels

Channels are used to communicate between goroutines...

package main

import (
    "fmt"
)

func sendData(ch chan string) {
    ch <- "Hello from goroutine"
}

func main() {
    ch := make(chan string)
    go sendData(ch)
    msg := <-ch
    fmt.Println(msg)
}

Module 8: Error Handling

8.1 Error Type

Errors in Go are represented using the error type...

package main

import (
    "fmt"
    "errors"
)

func divide(a int, b int) (int, error) {
    if b == 0 {
        return 0, errors.New("cannot divide by zero")
    }
    return a / b, nil
}

func main() {
    result, err := divide(10, 0)
    if err != nil {
        fmt.Println("Error:", err)
    } else {
        fmt.Println("Result:", result)
    }
}

Module 9: File I/O

9.1 Reading Files

Go provides the os and io/ioutil packages for file operations...

package main

import (
    "fmt"
    "io/ioutil"
)

func main() {
    data, err := ioutil.ReadFile("example.txt")
    if err != nil {
        fmt.Println("Error:", err)
        return
    }
    fmt.Println(string(data))
}

9.2 Writing Files

Writing to files can be done using the os package...

package main

import (
    "fmt"
    "os"
)

func main() {
    content := []byte("Hello, Go!")
    err := os.WriteFile("output.txt", content, 0644)
    if err != nil {
        fmt.Println("Error:", err)
    } else {
        fmt.Println("File written successfully")
    }
}
